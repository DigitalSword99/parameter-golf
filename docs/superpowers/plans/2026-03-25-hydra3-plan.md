```python
class TrigramHash(nn.Module):
    """Hash-based trigram features: hash(prev2, prev, cur) → learned embedding."""
    def __init__(self, num_buckets: int, dim: int):
        super().__init__()
        self.num_buckets = num_buckets
        self.embed = nn.Embedding(num_buckets, dim)
        nn.init.zeros_(self.embed.weight)

    def forward(self, input_ids: Tensor) -> Tensor:
        prev1 = F.pad(input_ids[:, :-1], (1, 0), value=0)
        prev2 = F.pad(input_ids[:, :-2], (2, 0), value=0)
        hashes = ((prev2.long() * 2654435761 + prev1.long() * 2246822519 + input_ids.long()) % self.num_buckets).long()
        return self.embed(hashes)
```

```python
class MoEMLP(nn.Module):
    """Mixture of Experts MLP with PHM-compressed experts and top-1 routing."""
    def __init__(self, dim: int, mlp_mult: int, num_experts: int = 8, phm_n: int = 4):
        super().__init__()
        hidden = mlp_mult * dim
        self.num_experts = num_experts
        self.router = nn.Linear(dim, num_experts, bias=False)
        # Each expert is a PHMLinear fc + proj pair
        self.expert_fc = nn.ModuleList([PHMLinear(dim, hidden, n=phm_n) for _ in range(num_experts)])
        self.expert_proj = nn.ModuleList([PHMLinear(hidden, dim, n=phm_n) for _ in range(num_experts)])
        for proj in self.expert_proj:
            proj._zero_init = True

    def forward(self, x: Tensor) -> tuple[Tensor, Tensor]:
        """Returns (output, aux_loss). aux_loss is load-balancing loss."""
        bsz, seq_len, dim = x.shape
        flat_x = x.reshape(-1, dim)  # (B*T, dim)
        # Router
        router_logits = self.router(flat_x)  # (B*T, num_experts)
        router_probs = F.softmax(router_logits, dim=-1)
        expert_idx = router_logits.argmax(dim=-1)  # (B*T,) top-1

        # Auxiliary load-balancing loss
        # fraction of tokens per expert (want uniform)
        tokens_per_expert = torch.zeros(self.num_experts, device=x.device)
        tokens_per_expert.scatter_add_(0, expert_idx, torch.ones_like(expert_idx, dtype=torch.float32))
        frac = tokens_per_expert / flat_x.size(0)
        avg_prob = router_probs.mean(dim=0)
        aux_loss = (frac * avg_prob).sum() * self.num_experts

        # Expert forward — loop over experts (simple, works with torch.compile)
        output = torch.zeros_like(flat_x)
        for e in range(self.num_experts):
            mask = (expert_idx == e)
            if mask.any():
                expert_input = flat_x[mask]
                h = F.leaky_relu(self.expert_fc[e](expert_input), negative_slope=0.5)
                expert_output = self.expert_proj[e](h.square())
                output[mask] = expert_output

        return output.reshape(bsz, seq_len, dim), aux_loss
```

```python
class DenseMLP3x(nn.Module):
    """Dense 3x expansion MLP — matches SOTA architecture."""
    def __init__(self, dim: int):
        super().__init__()
        self.fc = CastedLinear(dim, dim * 3, bias=False)
        self.proj = CastedLinear(dim * 3, dim, bias=False)
        self.proj._zero_init = True

    def forward(self, x: Tensor) -> tuple[Tensor, Tensor]:
        h = F.leaky_relu(self.fc(x), negative_slope=0.5)
        return self.proj(h.square()), torch.tensor(0.0, device=x.device)
```

```python
class WideMLP(nn.Module):
    """Project to wider dim, apply 2x MLP, project back."""
    def __init__(self, dim: int, wide_dim: int = 640):
        super().__init__()
        self.up = CastedLinear(dim, wide_dim, bias=False)
        self.fc = CastedLinear(wide_dim, wide_dim * 2, bias=False)
        self.proj = CastedLinear(wide_dim * 2, wide_dim, bias=False)
        self.down = CastedLinear(wide_dim, dim, bias=False)
        self.proj._zero_init = True

    def forward(self, x: Tensor) -> tuple[Tensor, Tensor]:
        x = self.up(x)
        h = F.leaky_relu(self.fc(x), negative_slope=0.5)
        x = self.down(self.proj(h.square()))
        return x, torch.tensor(0.0, device=x.device)
```

Note: All three heads return `(output, aux_loss)` for uniform interface. Only MoE has non-zero aux_loss.

Wait — this changes the Block interface too much and complicates the forward. Better approach: keep Block with an MLP, but make GPT own three SETS of MLPs (one per head) and swap which set is active.

**Revised approach — Block keeps its own attn but MLP is injected:**

Actually, the cleanest architecture: Block only does attention + residual. The GPT forward explicitly calls the selected head's MLP for each layer. This keeps Block simple and head selection in GPT.

```python
class Block(nn.Module):
    """Attention-only block. MLP is applied externally by GPT (head selection)."""
    def __init__(
        self,
        dim: int,
        num_heads: int,
        num_kv_heads: int,
        rope_base: float,
        qk_gain_init: float,
        phm_n: int = 4,
        layer_idx: int = 0,
    ):
        super().__init__()
        self.attn_norm = RMSNorm()
        self.mlp_norm = RMSNorm()
        self.attn = CausalSelfAttention(dim, num_heads, num_kv_heads, rope_base, qk_gain_init, phm_n=phm_n)
        self.attn_scale = nn.Parameter(torch.ones(dim, dtype=torch.float32))
        self.mlp_scale = nn.Parameter(torch.ones(dim, dtype=torch.float32))
        self.resid_mix = nn.Parameter(torch.stack((torch.ones(dim), torch.zeros(dim))).float())
        self.ln_scale = nn.Parameter(torch.tensor(1.0 / math.sqrt(layer_idx + 1), dtype=torch.float32))

    def forward_attn(self, x: Tensor, x0: Tensor, xsa: bool = False) -> Tensor:
        """Apply residual mix + attention only. Returns post-attention x."""
        mix = self.resid_mix.to(dtype=x.dtype)
        x = mix[0][None, None, :] * x + mix[1][None, None, :] * x0
        ln_s = self.ln_scale.to(dtype=x.dtype)
        attn_out = self.attn(self.attn_norm(x) * ln_s, xsa=xsa)
        return x + self.attn_scale.to(dtype=x.dtype)[None, None, :] * attn_out

    def forward_mlp(self, x: Tensor, mlp_out: Tensor) -> Tensor:
        """Apply MLP residual. mlp_out comes from whichever head was selected."""
        return x + self.mlp_scale.to(dtype=x.dtype)[None, None, :] * mlp_out

    def forward(self, x: Tensor, x0: Tensor, xsa: bool = False) -> Tensor:
        """Not used during Hydra-3 training — kept for compatibility."""
        raise NotImplementedError("Use forward_attn + forward_mlp separately for Hydra-3")
```

- [ ] **Step 1:** Rewrite GPT.__init__:

```python
class GPT(nn.Module):
    def __init__(
        self,
        vocab_size: int,
        num_layers: int,
        model_dim: int,
        num_heads: int,
        num_kv_heads: int,
        mlp_mult: int,
        tie_embeddings: bool,
        tied_embed_init_std: float,
        logit_softcap: float,
        rope_base: float,
        qk_gain_init: float,
        phm_n: int = 4,
        bigram_hash_buckets: int = 2048,
        trigram_hash_buckets: int = 4096,
        xsa_layers: int = 4,
        num_moe_experts: int = 8,
        head_b_mlp_mult: int = 3,
        head_c_wide_dim: int = 640,
    ):
        super().__init__()
        self.tie_embeddings = tie_embeddings
        self.tied_embed_init_std = tied_embed_init_std
        self.logit_softcap = logit_softcap
        self.num_layers = num_layers
        self.xsa_layers = xsa_layers
        self.tok_emb = nn.Embedding(vocab_size, model_dim)
        self.bigram_hash = BigramHash(bigram_hash_buckets, model_dim)
        self.trigram_hash = TrigramHash(trigram_hash_buckets, model_dim)

        # Shared attention backbone
        self.num_encoder_layers = num_layers // 2
        self.num_decoder_layers = num_layers - self.num_encoder_layers

        self.skip_map: dict[int, list[int]] = {}
        for dec_i in range(self.num_decoder_layers):
            sym_partner = self.num_encoder_layers - 1 - dec_i
            connections = []
            if 0 <= sym_partner < self.num_encoder_layers:
                connections.append(sym_partner)
            for enc_i in range(sym_partner - 4, -1, -4):
                connections.append(enc_i)
            self.skip_map[dec_i] = connections

        total_skip_connections = sum(len(v) for v in self.skip_map.values())
        self.skip_weights = nn.Parameter(torch.ones(total_skip_connections, model_dim, dtype=torch.float32))

        self.blocks = nn.ModuleList([
            Block(model_dim, num_heads, num_kv_heads, rope_base, qk_gain_init,
                  phm_n=phm_n, layer_idx=i)
            for i in range(num_layers)
        ])

        # Three MLP head sets (one per layer per head)
        self.head_a = nn.ModuleList([MoEMLP(model_dim, mlp_mult, num_moe_experts, phm_n) for _ in range(num_layers)])
        self.head_b = nn.ModuleList([DenseMLP3x(model_dim) for _ in range(num_layers)])
        self.head_c = nn.ModuleList([WideMLP(model_dim, head_c_wide_dim) for _ in range(num_layers)])
        self.heads = [self.head_a, self.head_b, self.head_c]

        self.final_norm = RMSNorm()
        self.lm_head = None if tie_embeddings else CastedLinear(model_dim, vocab_size, bias=False)
        self._init_weights()
```

- [ ] **Step 2:** Rewrite GPT.forward for head selection:

```python
    def forward(self, input_ids: Tensor, target_ids: Tensor, head_idx: int = -1) -> Tensor:
        """
        head_idx: 0=MoE, 1=Dense3x, 2=Wide. -1=ensemble all 3 (eval mode).
        Returns loss. During training, also includes MoE aux_loss.
        """
        x = self.tok_emb(input_ids)
        x = x + self.bigram_hash(input_ids)
        x = x + self.trigram_hash(input_ids)
        x = F.rms_norm(x, (x.size(-1),))
        x0 = x

        total_aux_loss = torch.tensor(0.0, device=x.device)
        encoder_outputs: list[Tensor] = []

        # Select which MLP heads to use
        if head_idx >= 0:
            active_heads = [self.heads[head_idx]]
        else:
            active_heads = self.heads  # all 3 for ensemble

        # Encoder
        for i in range(self.num_encoder_layers):
            use_xsa = i >= self.num_layers - self.xsa_layers
            x = self.blocks[i].forward_attn(x, x0, xsa=use_xsa)
            # MLP from active head(s)
            if len(active_heads) == 1:
                mlp_out, aux = active_heads[0][i](self.blocks[i].mlp_norm(x) * self.blocks[i].ln_scale.to(x.dtype))
                total_aux_loss = total_aux_loss + aux
            else:
                mlp_outs = []
                for head_set in active_heads:
                    mo, aux = head_set[i](self.blocks[i].mlp_norm(x) * self.blocks[i].ln_scale.to(x.dtype))
                    mlp_outs.append(mo)
                    total_aux_loss = total_aux_loss + aux
                mlp_out = sum(mlp_outs) / len(mlp_outs)
            x = self.blocks[i].forward_mlp(x, mlp_out)
            encoder_outputs.append(x)

        # Decoder with skips
        skip_w_idx = 0
        for dec_i in range(self.num_decoder_layers):
            eff_idx = self.num_encoder_layers + dec_i
            for enc_i in self.skip_map.get(dec_i, []):
                x = x + self.skip_weights[skip_w_idx].to(dtype=x.dtype)[None, None, :] * encoder_outputs[enc_i]
                skip_w_idx += 1
            use_xsa = eff_idx >= self.num_layers - self.xsa_layers
            x = self.blocks[eff_idx].forward_attn(x, x0, xsa=use_xsa)
            if len(active_heads) == 1:
                mlp_out, aux = active_heads[0][eff_idx](self.blocks[eff_idx].mlp_norm(x) * self.blocks[eff_idx].ln_scale.to(x.dtype))
                total_aux_loss = total_aux_loss + aux
            else:
                mlp_outs = []
                for head_set in active_heads:
                    mo, aux = head_set[eff_idx](self.blocks[eff_idx].mlp_norm(x) * self.blocks[eff_idx].ln_scale.to(x.dtype))
                    mlp_outs.append(mo)
                    total_aux_loss = total_aux_loss + aux
                mlp_out = sum(mlp_outs) / len(mlp_outs)
            x = self.blocks[eff_idx].forward_mlp(x, mlp_out)

        x = self.final_norm(x).reshape(-1, x.size(-1))
        targets = target_ids.reshape(-1)
        if self.tie_embeddings:
            logits_proj = F.linear(x, self.tok_emb.weight)
        else:
            logits_proj = self.lm_head(x)
        logits = self.logit_softcap * torch.tanh(logits_proj / self.logit_softcap)
        ce_loss = F.cross_entropy(logits.float(), targets, reduction="mean")
        return ce_loss + total_aux_loss * 0.01  # aux loss weight
```

```python
def estimate_hydra3_size(args) -> None:
    d = args.model_dim
    n = args.phm_n
    L = args.num_layers
    # Shared attention (PHM)
    kv_dim = args.num_kv_heads * (d // args.num_heads)
    def phm_p(d_in, d_out): return n**3 + n * (d_out//n) * (d_in//n)
    attn_per_layer = phm_p(d,d) + phm_p(d,kv_dim)*2 + phm_p(d,d)
    attn_total = attn_per_layer * L
    # Head A: MoE with PHM experts
    expert_per_layer = args.num_moe_experts * (phm_p(d, d*args.mlp_mult) + phm_p(d*args.mlp_mult, d))
    head_a_total = expert_per_layer * L + L * d * args.num_moe_experts  # + routers
    # Head B: Dense 3x
    head_b_total = (d * d * args.head_b_mlp_mult + d * args.head_b_mlp_mult * d) * L
    # Head C: Wide
    wd = args.head_c_wide_dim
    head_c_total = (d * wd + wd * wd * 2 + wd * 2 * wd + wd * d) * L
    # Embeddings + hash
    emb = args.vocab_size * d + args.bigram_hash_buckets * d + args.trigram_hash_buckets * d
    total = attn_total + head_a_total + head_b_total + head_c_total + emb + 5000
    compressed = total * 0.75 * 0.55 + 80_000
    print(f"Hydra-3 estimate: {total:,} params, ~{compressed / 1e6:.1f} MB compressed")
    if compressed > 15_500_000:
        print("WARNING: close to 16MB limit!")
```

### Task 12: Rewrite TTT for aggressive full-model Adam adaptation

**Modify:** `records/.../train_gpt.py` — eval_val_ttt function

```python
def eval_val_ttt(
    args,
    base_model: nn.Module,
    rank: int,
    world_size: int,
    device: torch.device,
    val_tokens: Tensor,
    base_bytes_lut: Tensor,
    has_leading_space_lut: Tensor,
    is_boundary_token_lut: Tensor,
) -> tuple[float, float]:
    """Aggressive Score-First TTT with Adam on all parameters."""
    chunk_size = args.ttt_chunk_size
    total_tokens = val_tokens.numel() - 1
    num_chunks = max(total_tokens // chunk_size, 1)

    # Adam on ALL parameters (not just S matrices)
    ttt_params = [p for p in base_model.parameters() if p.requires_grad]
    ttt_optimizer = torch.optim.Adam(ttt_params, lr=args.ttt_lr, betas=(0.9, 0.95))

    val_loss_sum = 0.0
    val_token_count = 0
    val_byte_count = 0.0

    for chunk_idx in range(num_chunks):
        start = chunk_idx * chunk_size
        end = min(start + chunk_size + 1, val_tokens.numel())
        chunk_tokens = val_tokens[start:end].to(device=device, dtype=torch.int64)

        # Phase 1: SCORE with ensemble (head_idx=-1)
        base_model.eval()
        with torch.inference_mode():
            num_seqs = (chunk_tokens.numel() - 1) // args.train_seq_len
            if num_seqs == 0:
                continue
            usable = num_seqs * args.train_seq_len
            x = chunk_tokens[:usable].reshape(num_seqs, args.train_seq_len)
            y = chunk_tokens[1:usable + 1].reshape(num_seqs, args.train_seq_len)
            with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                chunk_loss = base_model(x, y, head_idx=-1).item()

            prev_ids = x.reshape(-1)
            tgt_ids = y.reshape(-1)
            token_bytes = base_bytes_lut[tgt_ids].to(torch.float64)
            token_bytes += (has_leading_space_lut[tgt_ids] & ~is_boundary_token_lut[prev_ids]).to(torch.float64)
            val_loss_sum += chunk_loss * float(tgt_ids.numel())
            val_token_count += tgt_ids.numel()
            val_byte_count += token_bytes.sum().item()

        # Phase 2: ADAPT with aggressive Adam (skip last chunk)
        if chunk_idx < num_chunks - 1:
            base_model.train()
            for epoch in range(args.ttt_epochs):
                # Cosine LR decay per chunk
                epoch_lr = args.ttt_lr * 0.5 * (1 + math.cos(math.pi * epoch / args.ttt_epochs))
                for pg in ttt_optimizer.param_groups:
                    pg['lr'] = epoch_lr
                ttt_optimizer.zero_grad()
                with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
                    # Train with ensemble too (all heads benefit)
                    loss = base_model(x, y, head_idx=-1)
                loss.backward()
                torch.nn.utils.clip_grad_norm_(ttt_params, 1.0)
                ttt_optimizer.step()

    # Aggregate across ranks
    if dist.is_available() and dist.is_initialized():
        stats = torch.tensor([val_loss_sum, float(val_token_count), val_byte_count], device=device)
        dist.all_reduce(stats, op=dist.ReduceOp.SUM)
        val_loss_sum, val_token_count, val_byte_count = stats[0].item(), stats[1].item(), stats[2].item()

    val_loss = val_loss_sum / max(val_token_count, 1)
    bits_per_token = val_loss / math.log(2.0)
    tokens_per_byte = val_token_count / max(val_byte_count, 1)
    return val_loss, bits_per_token * tokens_per_byte
```

## Post-Implementation Notes

**torch.compile considerations for Hydra-3:**
- MoE expert loop (`for e in range(num_experts)`) is a static Python loop — torch.compile unrolls it. Safe with fullgraph=True as long as num_experts is a constant.
- `head_idx` changes per step. With torch.compile, this may cause recompilation for each head_idx value (0, 1, 2). After 3 compilations (one per head), subsequent calls hit cache. This adds ~30s to initial warmup but is free after.
- Ensemble mode (head_idx=-1) is a 4th compilation variant. Total: 4 compiled graphs cached.

**Memory estimate for 11L/512d:**
- Baseline (9L/512d) uses ~10GB per GPU
- Hydra-3 (11L/512d + 3 heads) stores more params but activations are the same as 11L/512d with one MLP head (only 1 head active per step). MoE top-1 routing means only 1 expert is computed per token.
- Estimated: ~12-15GB per GPU. Well within H100 80GB.

**If it OOMs anyway:**
- Reduce num_moe_experts from 8 to 4
- Reduce trigram_hash_buckets from 4096 to 2048
- These reduce stored params but not compute
